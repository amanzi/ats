import sys,os
import workflow.mesh

import numpy as np

# 2D transect and 2D mesh
x = np.arange(0,1000.01,10.)
z = 50.-0.05*x
m2 = workflow.mesh.Mesh2D.from_Transect(x,z)

# build up the layering structure as we extrude in the vertical
layer_dzs = [0.03, 0.03, 0.06, 0.12, 0.24, 0.48, 0.5, 0.54] + [2,]*10
layer_types = ["constant",]*len(layer_dzs)
layer_ncells = [1,]*len(layer_dzs)

# undisturbed mesh
layer_material_ids = [1001,]*(len(layer_dzs))
m3 = workflow.mesh.Mesh3D.extruded_Mesh2D(m2, layer_types, layer_dzs, 
                                        layer_ncells, layer_material_ids)
m3.write_exodus("hillslope2.exo")
