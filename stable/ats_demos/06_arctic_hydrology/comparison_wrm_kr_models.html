def readOBS(obs):
    filename = os.path.join('comparison_wrm_kr_models', obs)
    df = pd.read_table(filename, comment='#', delimiter=' ')
    df['discharge cumulation [mol/d]'] /= 55500
    df['surface-subsurface exchange cumulation [mol/d]'] /= -55500 
    df['thaw depth [m]'] *= 100
    df.rename(columns={'discharge cumulation [mol/d]': 'discharge [m3/d]',
                       'surface-subsurface exchange cumulation [mol/d]': 'infiltration [m3/d]',
                       'evaporation cumulation [m3/d]': 'evaporation [m3/d]',
                       'thaw depth [m]': 'thaw depth [cm]'}, inplace=True)
    df = df[365:].reset_index(drop=True)
    df['time [d]'] = df['time [d]'] - 365
    return df

models = ['wrm_vangenuchten-krel_vangenuchten', 'wrm_vangenuchten-krel_brookscorey',
          'wrm_vangenuchten-krel_sutraice', 'wrm_brookscorey-krel_brookscorey']
variables = ['discharge [m3/d]', 'infiltration [m3/d]', 'evaporation [m3/d]', 'thaw depth [cm]']
dict_combo = {}
for m in models:
    dict_combo[m] = readOBS('obs_hillslope_{}.dat'.format(m)).set_index('time [d]').to_dict('list')

reform = {(modelKey, varKey): values for modelKey, varDict in dict_combo.items() for varKey, values in varDict.items()}
df_combo = pd.DataFrame(reform)
df_combo['time [d]'] = range(0,365)

fig, ax = plt.subplots(4,1, figsize=(6,18))
plt.subplots_adjust(left=0.1, right=0.98, top=0.985, bottom=0.055, hspace=0.32)
for m in models:
    for i in range(4):
        df_combo.plot(x='time [d]', y=(m, variables[i]), ax=ax[i], label=m)

for i in range(4):
    ax[i].set_xlim(100, 250);
    ax[i].set_ylabel(variables[i])
    if i > 0:
        ax[i].get_legend().remove();
